各类设计模式

## 单例模式

  ```js
    const single = function (fn) {
      let ret = null
      return function() {
        ret || ret = fn.apply(this, arguments)
      }
    }
  ```

  标准单例模式：
  ```js
    cost Singleton = (name) => {
      this.name = name
      this.instance = null
    }

    Singleton.prototype.getName = function () {
      return this.name
    }

    Singleton.getInstance = (name) => {
      if (!this.instance) {
        this.instance = new Singleton(name)
      }
      return this.instance
    }
  ```
  缺点：不透明，我需要知道它是个单例，调用时，需要用 `Singleton.getInstance()` 的方式，而不是更常用的 `new` 方法

  透明的单例模式:

  ```js
    const Create = (name) => {
      this.name = name
    }
    const Singleton = (function() {
      var instance
      return function(name) {
        if (!instance) {
          instance = new Create(name) // create 为具体执行的方法
        }
        return instance
      }
    })()

    // 调用
    const user = new Singleton('allen')
    const user2 = new Singleton('allen')
  ```

  再优化：
  ```js

  const Singleton = (function () {
    var instance = null
    return function (fn) {
      if (!instance) {
        instance = fn.apply(this, arguments)
      }
      return instance
    }
  })()

  const Create = () => {
    this.name = 'allen'
  }

  const user = new Singleton(Create())
  const user2 = new Singleton(Create())
  
  ```

## 装饰者

  1. 原型链装饰者

    ```js
      Function.prototype.before = function (beforeFn) {
        var __self = this
        return () {
          beforeFn.apply(this, arguments)
          return __self.apply(this, arguments)
        }
      } 

      Function.prototype.after = function (afterFn) {
        var __self = this
        return () {
          __self.apply(this, arguments)
          return afterFn.apply(this, arguments)
        }
      }
    ```

  2. es6 中的 @ 操作符来实现装饰者
    * 类装饰者
    * 属性装饰者
    * 方法装饰者
